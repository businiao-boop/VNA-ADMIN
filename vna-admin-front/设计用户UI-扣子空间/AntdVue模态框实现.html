Promise风格Modal对话框实现

功能概述

基于Ant Design Vue组件库实现Promise风格的Modal对话框，提供可复用的promiseDialog函数，支持标准属性配置和Promise逻辑处理。

// 基本用法
promiseDialog({
    title: '确认操作',
    content: '您确定要执行此操作吗？'
}).then(result => {
    console.log(result); // { confirmed: true }
}).catch(error => {
    console.log(error); // { confirmed: false, error: "用户取消" }
});

实现代码

// promiseDialog.js
import { Modal } from 'ant-design-vue';

/**
 * Promise风格的Modal对话框
 * @param {Object} config 配置参数
 * @returns {Promise} 返回Promise对象
 */
export function promiseDialog(config = {}) {
    return new Promise((resolve, reject) => {
        let formData = {};
        let modalRef = null;
        
        const defaultConfig = {
            title: '提示',
            content: '',
            width: 520,
            okText: '确定',
            cancelText: '取消',
            centered: false,
            maskClosable: true,
            onOk: () => {
                resolve({ confirmed: true, data: formData });
                modalRef.destroy();
            },
            onCancel: () => {
                reject({ confirmed: false, error: "用户取消" });
                modalRef.destroy();
            }
        };
        
        const mergedConfig = { ...defaultConfig, ...config };
        
        // 处理自定义内容
        if (config.content) {
            mergedConfig.content = config.content;
        }
        
        // 创建Modal实例
        modalRef = Modal.confirm(mergedConfig);
        
        // 提供关闭方法
        modalRef.close = () => {
            modalRef.destroy();
            reject({ confirmed: false, error: "程序关闭" });
        };
        
        // 更新表单数据
        modalRef.updateFormData = (data) => {
            formData = { ...formData, ...data };
        };
    });
}
// 使用示例
// 1. 基础确认框
promiseDialog({
    title: '确认删除',
    content: '您确定要删除此项吗？'
}).then(result => {
    if (result.confirmed) {
        // 执行删除操作
    }
});

// 2. 带表单的对话框
promiseDialog({
    title: '填写信息',
    content: `
        <a-input placeholder="请输入姓名" @change="e => updateFormData({ name: e.target.value })" />
        <a-input placeholder="请输入年龄" @change="e => updateFormData({ age: e.target.value })" />
    `,
    onOk: () => {
        // 表单验证逻辑
        if (!formData.name) {
            return Promise.reject('请输入姓名');
        }
        return Promise.resolve();
    }
}).then(result => {
    console.log('表单数据:', result.data);
});

// 3. 自定义对话框
promiseDialog({
    title: '自定义对话框',
    content: '这是一个自定义样式的对话框',
    width: 800,
    okText: '我同意',
    cancelText: '我拒绝',
    okButtonProps: {
        type: 'danger'
    },
    cancelButtonProps: {
        type: 'dashed'
    }
});
